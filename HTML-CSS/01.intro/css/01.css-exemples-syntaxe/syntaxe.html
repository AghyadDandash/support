SYNTAXE CSS

			
HTML selectors

      
Cibler les balises HTML


			
			body {
				color: #1a1a1a;
				background-color: #e5e5e5;
				font-size: 16px;
			}

			h1 {
				font-size: 3em;
				line-height: 2.5;
				text-transform: uppercase;
				text-align: center /* Pour le dernier élément on peut ignorer le ";" */
			}

			h2 {
				font-size: 2em;
			}

			pre {
				display: block;
				width: 60%;
				margin: 0 auto 30px	;
				padding: 20px;
				background-color: white;
				white-space: pre-wrap;
				tab-size: 2;
			}


			nav {
				white-space: normal;
			}

			quote {
				color: #aaa;
			}

			//Quand plusieurs balises partagent les mêmes styles
			h1, h2 {
				margin: 0;
				font-weight: normal;
				color: #212121;
			}

			//Relation parent - enfant(s)
			pre code {
				font-family: monospace;
				font-size: 1.2em;
			}

			code p {
				color: darkgrey;
			}

			nav ul {
				font-weight: normal;
			}

			nav ul li {
				color: #7a7a7a;
			}

			//Relation parent - enfant, changer seulement le premier enfant
			nav > ul {
				font-weight: bold;
			}

			nav > ul > li {
				color: #212121;
			}

			//Relation parent - enfant, changer seulement le dernier enfant
			code p:last-child {
				color: orange;
			}

			//Relation parent - enfant, ignorer les deux premiers enfants
			code p + p + p {
				color: #212121;
			}


			//Changer les styles en "hover" - survol
			nav li:hover {
				text-indent: 20px;
			}

			nav li ul li:hover  {
				text-indent: 5px;
			}
			
		
			
HTML attribute "class"
		 Apliquer les styles seulement sur les éléments avec ".nom"


			
			.nowrap {
				white-space: nowrap;
			}

			.html-color {
				color: navy;
			}

			.css-color {
				color: fuchsia;
			}

			code .last-p {
				background-color: #f5f5f5;
			}
			
		
			
HTML attribute "id"
		 Apliquer les styles seulement sur les éléments avec "#nom"


			
			#nav {

			}

			#menu-1 {
				background-color: #e5e5e5;
			}

			#menu-2 {
				background-color: #f5f5f5;
			}

			code #middle-p {
				background-color: #fc0;
			}
